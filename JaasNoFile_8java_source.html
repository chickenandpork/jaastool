175  //context = new LoginContext(name, callbackHandler);

177  context = new LoginContext(processName, null /* subject */, callbackHandler, new NoConfFileConfiguration(domain, principal, server, password, debug));

268  public void handle(Callback[] callbacks) throws UnsupportedCallbackException
269  {

275  }

281  }
282  }
